-- Exploratory Data Analysis 

SELECT * FROM walmart_sales;

-- 1. Total Sales, Total Revenue and Profit

SELECT count(*) AS Total_Sales, SUM(total) AS Total_Revenue, 
SUM(total*profit_margin) AS Total_Profit
FROM walmart_sales;

-- 2. Find  number_of_transaction and number_of_quantity sold
-- based on payment method

SELECT payment_method, COUNT(*) AS no_of_transaction, SUM(quantity) AS no_of_quantity FROM
walmart_sales GROUP BY payment_method;

-- 3. Identify the highest average-rated category in each branch

select * from
(
SELECT Branch, category, cast(AVG(rating) AS DECIMAL(10,2)) AS average_rating,
RANK() OVER(PARTITION BY Branch ORDER BY AVG(rating) DESC) AS ranking
FROM walmart_sales GROUP BY Branch, category
) x
WHERE ranking = 1;

-- 4. Identify Sales based on Month

SELECT EXTRACT(MONTH FROM date)  AS Month,
CAST(SUM(total) AS DECIMAL(10,2)) AS Monthly_Sales
FROM walmart_sales
GROUP BY Month
ORDER BY Monthly_Sales DESC;

-- 5. Find the busiest day for each branch on the number of transactions

SELECT * FROM walmart_sales;

SELECT * FROM
(
SELECT Branch, 
DAYNAME(date) AS day_name,
count(*) as no_of_transactions,
RANK() OVER(PARTITION BY Branch ORDER BY COUNT(*) DESC) AS ranking
FROM walmart_sales
GROUP BY Branch, day_name
) x
WHERE ranking = 1;

-- 6. Find the total number of sales, total revenue and total profit based on Branch

SELECT Branch, COUNT(*) AS Number_of_Sales,
CAST(SUM(total) AS DECIMAL(10,2)) AS Total_Revenue,
CAST(SUM(total*profit_margin) AS DECIMAL(10,2)) AS Profit_Value
FROM walmart_sales
GROUP BY Branch
ORDER BY Branch;

-- 7. Find the total number of sales, total revenue and total profit based on category  

SELECT category, 
COUNT(*) AS Total_Sales,
CAST(SUM(total) AS DECIMAL(10,2)) AS total_revenue,
CAST(SUM(total * profit_margin) AS DECIMAL(10,2)) AS total_profit
FROM walmart_sales
GROUP BY category
ORDER BY total_profit DESC;

-- 8. Calculate number of transactions by payment method based on city

SELECT city, payment_method, COUNT(*) AS no_of_transactions FROM
walmart_sales GROUP BY city,payment_method
ORDER BY city;

-- 9. Determine the average, minimum and maximum rating of products for each Branch and city, category

SELECT Branch, City, category,
CAST(AVG(rating) AS DECIMAL(5,2)) AS average_rating, 
MIN(rating) AS minimum_rating,
MAX(rating) AS maximum_rating
FROM walmart_sales
GROUP BY Branch, City, category
ORDER BY Branch;

-- 10. Determine the most common payment method for each Branch.

WITH payment as
(
SELECT Branch, payment_method,
COUNT(*) AS total_transaction,
RANK() OVER(PARTITION BY Branch ORDER BY COUNT(*) DESC) AS ranking
FROM walmart_sales
GROUP BY Branch, payment_method
)
SELECT * FROM payment 
WHERE ranking = 1;

-- 11. Calculate the revenue percentage by Branch based on Total Revenue

SELECT Branch,
CAST(((Total_Revenue/(SELECT SUM(total) FROM walmart_sales))*100) AS DECIMAL(5,2)) AS Revenue_Percentage
FROM
(SELECT Branch, SUM(total) as Total_Revenue
FROM walmart_sales
GROUP BY Branch) as x
ORDER BY Branch;

-- 12. For each branch, find the hour of the day with the highest sales

SELECT * FROM (
SELECT Branch, EXTRACT(HOUR FROM time) AS Hour, COUNT(*) AS Total_Sales,
RANK() OVER(PARTITION BY Branch ORDER BY COUNT(*) DESC) as Sales_Rank
FROM walmart_sales
GROUP BY Branch, EXTRACT(HOUR FROM time)
ORDER BY Branch) AS x
WHERE Sales_Rank = 1;

-- 13. Identify the category that gives the highest profit in each city

SELECT * FROM 
(
SELECT city, category, CAST(SUM(total*profit_margin) AS DECIMAL(10,2)) AS Profit_Value,
RANK() OVER(PARTITION BY city ORDER BY(SUM(total*profit_margin)) DESC) AS Profit_Rank
FROM walmart_sales
GROUP BY city, category
ORDER BY Profit_Rank) AS x
WHERE Profit_Rank = 1;

-- 14. Month over month sales by category

 SELECT category, EXTRACT(MONTH FROM date) AS Month,
 CAST(SUM(total) AS DECIMAL(10,2)) AS Monthly_Sales,
 CAST(LAG(SUM(total)) OVER (PARTITION BY category ORDER BY EXTRACT(MONTH FROM date)) AS DECIMAL(10,2)) AS Prev_Month_Sales,
 CAST((SUM(total) - LAG(SUM(total)) OVER (PARTITION BY category ORDER BY EXTRACT(MONTH FROM date))) / 
 NULLIF(LAG(SUM(total)) OVER (PARTITION BY category ORDER BY EXTRACT(MONTH FROM date)),0) * 100 AS DECIMAL (10,2)) AS MoM_Sales_Percentage
 FROM walmart_sales
 GROUP BY Month, category;

-- 15. Year over Year Sales

 SELECT EXTRACT(YEAR FROM date) AS Year,
 CAST(SUM(total) AS DECIMAL(10,2)) AS Yearly_Sales,
 CAST(LAG(SUM(total)) OVER (ORDER BY EXTRACT(YEAR FROM date)) AS DECIMAL(10,2)) AS Prev_Year_Sales,
 CAST((SUM(total) - LAG(SUM(total)) OVER (ORDER BY EXTRACT(YEAR FROM date))) / 
 NULLIF(LAG(SUM(total)) OVER (ORDER BY EXTRACT(YEAR FROM date)),0) * 100 AS DECIMAL(10,2)) AS YoY_Growth_Percentage
 FROM walmart_sales
 GROUP BY Year;
 
-- 16. Year over year growth for each Branch
 
 SELECT Branch, EXTRACT(YEAR FROM date) AS Year,
 CAST(SUM(total) AS DECIMAL(10,2)) AS Yearly_Sales,
 CAST(LAG(SUM(total)) OVER (PARTITION BY Branch ORDER BY EXTRACT(YEAR FROM date)) AS DECIMAL(10,2)) AS Prev_Year_Sales,
 CAST((SUM(total) - LAG(SUM(total)) OVER (PARTITION BY Branch ORDER BY EXTRACT(YEAR FROM date))) / 
 NULLIF(LAG(SUM(total)) OVER (PARTITION BY Branch ORDER BY EXTRACT(YEAR FROM date)),0) * 100 AS DECIMAL(10,2)) AS YoY_Growth_Percentage
 FROM walmart_sales
 GROUP BY Year, Branch;
